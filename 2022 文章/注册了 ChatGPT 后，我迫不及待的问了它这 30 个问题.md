function debounce(fn, delay) {
  let timer = null;

function throttle(fn, delay) {
  let timer = null;

  return function() {
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(this, arguments);

        timer = null; // 执行完成后，重置为 null，以便下次触发事件时，再次执行函数。
